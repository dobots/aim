IF(WIN32)
cmake_minimum_required(VERSION 2.6)
ELSE(WIN32)
cmake_minimum_required(VERSION 2.4)
ENDIF(WIN32)

# The directory with all the FindXXX modules
SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake/")

SET(PROJECT_MNAME "SumModule")
SET(PROJECT_DIR "sum")

# The macro below forces the build directory to be different from source directory:
INCLUDE( MacroEnsureOutOfSourceBuild )
MACRO_ENSURE_OUT_OF_SOURCE_BUILD(
	"${PROJECT_MNAME} requires an out of source build."
)

# Build options
OPTION(BUILD_YARP      "Build module with YARP" OFF)
OPTION(BUILD_ROS       "Build module with ROS" OFF)

# Start a project
PROJECT(${PROJECT_MNAME})

# Find all cmake packages
IF (BUILD_YARP)
  FIND_PACKAGE(YARP REQUIRED)
ENDIF (BUILD_YARP)

IF (BUILD_ROS)
  include($ENV{ROS_ROOT}/core/rosbuild/rosbuild.cmake)
  rosbuild_init()
ENDIF (BUILD_ROS)

# Header files
IF (BUILD_YARP)
  INCLUDE_DIRECTORIES(${YARP_INCLUDE_DIRS})
ENDIF (BUILD_YARP)

# Shared libraries
IF (BUILD_YARP)
  SET(LIBS ${LIBS} ${YARP_LIBRARIES})
ENDIF (BUILD_YARP)

# Turn on testing.
#enable_testing()

# Some debug information
MESSAGE("${PROJECT_MNAME} is using CXX flags: ${CMAKE_CXX_FLAGS}")
MESSAGE ("Libraries included: ${LIBS}")

# Source directory
SET(EXAMPLE_DIR "example")

# Search for source code.
FILE(GLOB folder_source ${EXAMPLE_DIR}/${PROJECT_DIR}/*.cpp ${EXAMPLE_DIR}/${PROJECT_DIR}/*.cc ${EXAMPLE_DIR}/${PROJECT_DIR}/*.c)
FILE(GLOB folder_header ${EXAMPLE_DIR}/${PROJECT_DIR}/*.h)
SOURCE_GROUP("Source Files" FILES ${folder_source})
SOURCE_GROUP("Header Files" FILES ${folder_header})

# Automatically add include directories if needed.
FOREACH(header_file ${folder_header})
  GET_FILENAME_COMPONENT(p ${header_file} PATH)
  INCLUDE_DIRECTORIES(${p})
ENDFOREACH(header_file ${folder_header})

#add_subdirectory(example)
#add_subdirectory(harness)

#rosbuild_add_executable(${PROJECT_MNAME} ${folder_source} ${folder_header})

# Set up our main executable.
IF (folder_source)
   ADD_EXECUTABLE(${PROJECT_MNAME} ${folder_source} ${folder_header})
   TARGET_LINK_LIBRARIES(${PROJECT_MNAME} ${LIBS})
   install(TARGETS ${PROJECT_MNAME} RUNTIME DESTINATION bin)   
ELSE (folder_source)
  MESSAGE(FATAL_ERROR "No source code files found. Please add something")
ENDIF (folder_source)
